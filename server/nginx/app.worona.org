server {
    listen 80;
    listen 443 ssl;

    # SSL cert
    ssl_certificate /etc/nginx/ssl/nginx.crt;
    ssl_certificate_key /etc/nginx/ssl/nginx.key;

    # Disable SSLv3 (Poodle bug)
    ssl_protocols TLSv1 TLSv1.1 TLSv1.2;

    root /var/www/worona-cdn/packages/dist/core-app-worona/app/prod/html;
    index index.html;
    server_name app.worona.org;

    # No Cache
    expires epoch;

    location / {
  		try_files $uri $uri/ /index.html;
  	}
}
